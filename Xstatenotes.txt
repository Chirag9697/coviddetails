1. Install Xstate
-> npm install xstate

2. Install Xstate/react
-> npm install @xstate/react

3. Create a file name called counterMachine
-> import {createMachine, assign} from 'xstate'
    export const counterMachine =     createMachine({
    	context:{
            count: 0, //Initialize the value
        }
        on:{
            Increment:{
                actions:assign({
                    count:(context)=>context.count + 1
                }),
            },
        },
    }) 

//Machine Ready
//Now import in react component

1. import {useMachine} from '@xstate/react'
2. import counterMachine from './counterMachine'
3. add state in useMachine
    const [state,send]=useMachine(counterMachine)
4. use it in jsx
    <h1>{state.context.count}</h1>
